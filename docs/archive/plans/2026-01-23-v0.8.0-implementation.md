```javascript
/**
 * Generate Executive signature style
 * Target: VPs, C-Suite, Senior Leadership
 * Design: Large name, center-aligned, minimal info, no social media
 * @param {Object} data - Form data
 * @param {Object} socialOptions - Social media config (ignored for Executive)
 * @param {string} accentColor - Accent color for subtle line
 * @returns {string} HTML signature
 */
generateExecutive(data, socialOptions, accentColor = '#E42527') {
    const darkModeStyles = this.getDarkModeStyles();
    const logos = this.getLogoUrls();
    const websiteUrl = data.website || 'https://www.zoho.com';

    // Executive template has NO social media

    return `
${darkModeStyles}
<table cellpadding="0" cellspacing="0" border="0" style="font-family: Arial, sans-serif; font-size: 14px; line-height: 1.6; max-width: 500px; margin: 0 auto;">
    <tr>
        <td style="padding: 20px 0; text-align: center;">
            <!-- Name (large, bold) -->
            <div style="font-size: 20px; font-weight: 700; color: #000000; margin-bottom: 4px;" class="sig-name">
                ${this.escapeHtml(data.name)}
            </div>

            <!-- Accent line below name -->
            <div style="width: 60px; height: 1px; background: ${accentColor}; margin: 8px auto 12px auto;"></div>

            <!-- Title (if provided) -->
            ${data.title ? `
            <div style="font-size: 14px; color: #666666; margin-bottom: 12px;" class="sig-title">
                ${this.escapeHtml(data.title)}
            </div>
            ` : ''}

            <!-- Logo -->
            <div style="margin: 16px 0;">
                <a href="${websiteUrl}" style="text-decoration: none; display: inline-block;">
                    <img src="${logos.light}"
                         alt="Zoho"
                         class="sig-logo-light"
                         style="height: 40px; display: block; border: 0;"
                         height="40">
                    <img src="${logos.dark}"
                         alt="Zoho"
                         class="sig-logo-dark"
                         style="height: 40px; display: none; border: 0;"
                         height="40">
                </a>
            </div>

            <!-- Contact info (vertical stack, centered) -->
            ${data.email ? `
            <div style="margin-bottom: 4px;">
                <a href="mailto:${data.email}" style="color: ${accentColor}; text-decoration: none;" class="sig-link">
                    ${this.escapeHtml(data.email)}
                </a>
            </div>
            ` : ''}

            ${data.phone ? `
            <div style="margin-bottom: 4px;">
                <a href="tel:${this.sanitizePhone(data.phone)}" style="color: ${accentColor}; text-decoration: none;" class="sig-link">
                    ${this.escapeHtml(data.phone)}
                </a>
            </div>
            ` : ''}

            ${data.website ? `
            <div style="margin-bottom: 4px;">
                <a href="${data.website}" style="color: ${accentColor}; text-decoration: none;" class="sig-link">
                    zoho.com
                </a>
            </div>
            ` : ''}
        </td>
    </tr>
</table>
`.trim();
},
```

### Step 3: Implement generateBold function with contrast-aware text color

**Action:** Add new function after `generateExecutive()`

```javascript
/**
 * Generate Bold signature style
 * Target: Marketing, Events Team
 * Design: Colored name/title block, eye-catching
 * @param {Object} data - Form data
 * @param {Object} socialOptions - Social media config
 * @param {string} accentColor - Background color for name block
 * @returns {string} HTML signature
 */
generateBold(data, socialOptions, accentColor = '#E42527') {
    const darkModeStyles = this.getDarkModeStyles();
    const logos = this.getLogoUrls();
    const websiteUrl = data.website || 'https://www.zoho.com';

    // Determine text color based on background (yellow needs dark text)
    const textColor = accentColor === '#F9B21D' ? '#333333' : '#FFFFFF';

    // Build contact links
    const contacts = [];
    if (data.email) {
        contacts.push(`<a href="mailto:${data.email}" style="color: ${accentColor}; text-decoration: none;" class="sig-link">${this.escapeHtml(data.email)}</a>`);
    }
    if (data.phone) {
        contacts.push(`<a href="tel:${this.sanitizePhone(data.phone)}" style="color: ${accentColor}; text-decoration: none;" class="sig-link">${this.escapeHtml(data.phone)}</a>`);
    }
    if (data.linkedin) {
        contacts.push(`<a href="${data.linkedin}" style="color: ${accentColor}; text-decoration: none;" class="sig-link">LinkedIn</a>`);
    }
    if (data.website) {
        contacts.push(`<a href="${data.website}" style="color: ${accentColor}; text-decoration: none;" class="sig-link">zoho.com</a>`);
    }

    // Build social media links if enabled
    let socialHtml = '';
    if (socialOptions.enabled && socialOptions.channels && socialOptions.channels.length > 0) {
        const socialLinks = socialOptions.channels.map(channel => {
            const channelData = this.SOCIAL_CHANNELS[channel];
            if (!channelData) return '';

            return `<a href="${channelData.url}" style="display: inline-block; margin: 0 4px; width: 24px; height: 24px; background: ${accentColor}; color: ${textColor}; text-align: center; line-height: 24px; text-decoration: none; border-radius: 4px; font-size: 12px; font-weight: 600;" class="sig-link">${channelData.icon}</a>`;
        }).filter(link => link !== '').join('');

        if (socialLinks) {
            socialHtml = `
            <div style="margin-top: 8px;">
                ${socialLinks}
            </div>
            `;
        }
    }

    return `
${darkModeStyles}
<table cellpadding="0" cellspacing="0" border="0" style="font-family: Arial, sans-serif; font-size: 14px; line-height: 1.6; max-width: 500px;">
    <tr>
        <td style="padding: 12px 0;">
            <table cellpadding="0" cellspacing="0" border="0" style="width: 100%;">
                <tr>
                    <!-- Logo (left) -->
                    <td style="width: 60px; vertical-align: middle; padding-right: 12px;">
                        <a href="${websiteUrl}" style="text-decoration: none; display: block;">
                            <img src="${logos.light}"
                                 alt="Zoho"
                                 class="sig-logo-light"
                                 style="height: 32px; display: block; border: 0;"
                                 height="32">
                            <img src="${logos.dark}"
                                 alt="Zoho"
                                 class="sig-logo-dark"
                                 style="height: 32px; display: none; border: 0;"
                                 height="32">
                        </a>
                    </td>

                    <!-- Info block -->
                    <td style="vertical-align: middle;">
                        <!-- Colored name/title block -->
                        <div style="background: ${accentColor}; color: ${textColor}; padding: 12px 16px; border-radius: 8px; margin-bottom: 8px;">
                            <div style="font-size: 16px; font-weight: 700; margin-bottom: 2px;">
                                ${this.escapeHtml(data.name)}
                            </div>
                            ${data.title ? `
                            <div style="font-size: 13px; opacity: 0.9;">
                                ${this.escapeHtml(data.title)}
                            </div>
                            ` : ''}
                        </div>

                        <!-- Contact info (horizontal) -->
                        ${contacts.length > 0 ? `
                        <div style="font-size: 13px; color: #666666;">
                            ${contacts.join('<span style="margin: 0 6px; color: #CCCCCC;" class="sig-separator">•</span>')}
                        </div>
                        ` : ''}

                        <!-- Social media -->
                        ${socialHtml}
                    </td>
                </tr>
            </table>
        </td>
    </tr>
</table>
`.trim();
},
```
